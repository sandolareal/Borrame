# Borrame

Hola Cruel Mundo !

# Traer la última versión de la rama
git pull 

# Subir la ultima versión de la rama al servidor
git push

# Sincronizar las ramas del servidor con la local
git fetch

# Cambiar de rama
git checkout <nombre_rama>

# crear una rama nueva
git checkout -b <nombre_rama>
git branch <nombre_rama>

# listar todas las ramas
git branch --all
